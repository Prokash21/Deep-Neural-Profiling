# STEP 1: Create a new conda environment with Python 3.11
conda create -n myenv python=3.11 -y

# STEP 2: Activate the environment
conda activate myenv

# STEP 3: Install ipykernel so it appears in Jupyter
conda install ipykernel -y

# STEP 4: Add this environment to Jupyter
python -m ipykernel install --user --name=myenv --display-name "Python 3 (myenv)"

# STEP 5: Launch Jupyter Notebook
jupyter notebook
